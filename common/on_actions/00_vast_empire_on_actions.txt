on_monthly_pulse = {
	events = {
		# 1860 Escalation
		vast_empire.2
	}
	
	trigger = {
		any_country = {
			tag = GBR
			has_country_modifier = vast_empire_GBR_base
			date = 1860.1.1  # Fires on or after Jan 1, 1860
		}
	}
}

# Event for 1860 escalation (in events/vast_empire_events.txt)
country_event = {
	id = vast_empire.2
	title = "The Empire Expands Further"
	desc = "By 1860, the strains of our vast empire have intensified, raising military costs even higher."
	
	trigger = {
		tag = GBR
		has_country_modifier = vast_empire_GBR_base
		NOT = { has_country_modifier = vast_empire_GBR_1860 }
		date >= 1860.1.1
	}
	
	immediate = {
		# Remove old modifier
		remove_country_modifier = vast_empire_GBR_base
		# The hidden one stays
	}
	
	option = {
		name = "The burden grows..."
		add_country_modifier = {
			name = vast_empire_GBR_1860
			days = -1
		}
	}
}

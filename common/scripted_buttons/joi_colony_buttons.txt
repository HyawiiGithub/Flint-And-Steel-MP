	name = "berlin_conference_proposal_1"
	desc = "berlin_conference_proposal_1_desc"

	visible = {
		exists = c:GBR
		c:GBR = { has_journal_entry = berlin_conference }
	}

	possible = {
		custom_tooltip = {
		text = berlin_conference_proposal_possible_tt
		NOT = { has_variable = you_have_voted_in_berlin_conference }
		}
		custom_tooltip = {
		text = every_nation_has_sign_last_treaty_tt
		you_can_now_start_to_vote_trigger = yes
		}
		custom_tooltip = {
		text = not_winner_of_wf_or_cf_tt
		trigger_if = {
				limit = { has_journal_entry = berlin_conference_congo }
				c:GBR = {
					NOR = {
					has_variable = winner_of_central_africa
					has_variable = winner_of_west_africa
					}
				}
			}
		}
		custom_tooltip = {
		text = not_winner_of_con_2_tt
		trigger_if = {
				limit = { has_journal_entry = berlin_conference_east_africa }
				c:GBR = {
					NOT = {
					has_variable = winner_of_congo
					}
				}
			}
		}
		custom_tooltip = {
		text = not_winner_of_an_or_moz_or_con_tt
		trigger_if = {
				limit = { has_journal_entry = berlin_conference_south_africa }
				c:GBR = {
					NOR = {
					has_variable = winner_of_angola
					has_variable = winner_of_congo
					has_variable = winner_of_mozambique
					}
				}
			}
		}
		custom_tooltip = {
		text = winner_of_an_or_moz_or_sf_or_ef_tt
		trigger_if = {
				limit = { has_journal_entry = berlin_conference_zambia }
				c:GBR = {
					OR = {
					has_variable = winner_of_angola
					has_variable = winner_of_mozambique
					has_variable = winner_of_east_africa
					has_variable = winner_of_south_africa
					}
				}
			}
		}
	}

	ai_chance = {
		value = 10
		if = {
			limit = { 
				OR = {
					has_journal_entry = berlin_conference_niger
					has_journal_entry = berlin_conference_kenya
					has_journal_entry = berlin_conference_south_africa
					has_journal_entry = berlin_conference_zambia
					}
				OR = {
					country_has_primary_culture = cu:british
					country_has_primary_culture = cu:north_german
					country_has_primary_culture = cu:french
					country_has_primary_culture = cu:wallonian
					country_has_primary_culture = cu:yankee
					country_has_primary_culture = cu:portuguese
					country_has_primary_culture = cu:south_german
				}
			}
			add = 1000
		}
		if = {
			limit = { 
				NOT = { THIS ?= c:GBR }
				c:GBR.relations:root >= 100
			}
			add = 10
		}
		if = {
			limit = { 
				NOT = { THIS ?= c:GBR }
				c:GBR.relations:root >= 80
			}
			add = 10
		}
		if = {
			limit = { 
				NOT = { THIS ?= c:GBR }
				c:GBR.relations:root >= 60
			}
			add = 10
		}
		if = {
			limit = { 
				NOT = { THIS ?= c:GBR }
				c:GBR.relations:root >= 40
			}
			add = 10
		}
		if = {
			limit = { 
				NOT = { THIS ?= c:GBR }
				c:GBR.relations:root <= -20
			}
			add = -10
		}
		if = {
			limit = { 
				NOT = { THIS ?= c:GBR }
				c:GBR.relations:root <= -40
			}
			add = -10
		}
		if = {
			limit = { 
				NOT = { THIS ?= c:GBR }
				c:GBR.relations:root <= -60
			}
			add = -10
		}
		if = {
			limit = { 
				NOT = { THIS ?= c:GBR }
				c:GBR.relations:root <= -80
			}
			add = -100
		}
		if = {
			limit = { 
				NOT = { THIS ?= c:GBR }
				c:GBR = {
					infamy >= 50
				}
			}
			add = -10
		}
		if = {
			limit = { 
				NOT = { THIS ?= c:GBR }
				c:GBR = {
					infamy >= 75
				}
			}
			add = -10
		}
		if = {
			limit = { 
				NOT = { THIS ?= c:GBR }
				c:GBR = {
					infamy >= 100
				}
			}
			add = -100
		}
	}

	effect = {
		change_relations = {
			country = c:GBR
			value = 10
		}
		if = {
			limit = { country_rank >= rank_value:great_power }
			change_global_variable = { name = berilin_conference_support_gbr_proposal add = 2 }
		}
		if = {
			limit = { country_rank <= rank_value:major_power }
			change_global_variable = { name = berilin_conference_support_gbr_proposal add = 1 }
		}
		if = {
			limit = { check_historical_claim_proposal_weight = yes }#历史宣称添加1票权
			change_global_variable = { name = berilin_conference_support_gbr_proposal add = 1 }
		}
		if = {
			limit = { check_uti_possidetis_proposal_weight = yes }#有效占领原则
			change_global_variable = { name = berilin_conference_support_gbr_proposal add = 100 }
		}
		set_variable = you_have_voted_in_berlin_conference
	}
}

			value = 0
			}
		}

		ig:ig_intelligentsia = {
			add_ruling_interest_group = yes
		}
		trigger_event = { id = joi_flavor_chi.1 }
		add_journal_entry = { 
			type = western_diplomatic_affairs_movement
		}

			type = western_industry_affairs_movement

			type = western_railway_affairs_movement
		}
		add_journal_entry = { 
			type = western_army_affairs_movement
		}
		add_journal_entry = { 
			type = western_navy_affairs_movement
		}
		add_journal_entry = { 
			type = western_education_affairs_movement
		}
    }

    on_monthly_pulse = {
    	random_events = {
		85 = 0
		15 = joi_flavor_chi.4
		}
    }

    modifiers_while_active = {
		chi_western_affairs_movement
	}

	complete = {
		custom_tooltip = {
			text = western_affairs_movement_tt
			var:western_affairs_movement_var = 6
		}
	}

	on_complete = {
		trigger_event = {
			id = joi_flavor_chi.2
		}
		set_variable = complete_western_affairs_movement_var
	}

	fail = {
		ig:ig_landowners = { is_in_government = no }
		ig:ig_intelligentsia = { 
			is_in_government = no 
			is_marginal = yes
		}
	}

	on_fail = {
		trigger_event = { id = joi_flavor_chi.3 }
		set_variable = western_affairs_movement_failed
	}

	invalid = {
		any_civil_war = {
			is_civil_war_type = revolution
			civil_war_progress >= 0.75
			} 
	}

	on_timeout = {
		set_variable = western_affairs_movement_failed
	}

	current_value = {
		value = var:western_affairs_movement_var
	}

	goal_add_value = {
		add = 6
	}

	status_desc = {
		triggered_desc = { 
			desc = western_affairs_movement_status
		}
	}

	timeout = 18250
	
	progressbar = yes

	weight = 100

	should_be_pinned_by_default = yes
}

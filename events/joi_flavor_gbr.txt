joi_flavor_gbr.1 = {#不列颠治世
	type = country_event
	placement = root
	title = joi_flavor_gbr.1.t
	desc = joi_flavor_gbr.1.d
	flavor = joi_flavor_gbr.1.f

	duration = 3

	event_image = {
		video = "unspecific_world_fair"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	trigger = {
	}

	immediate = {
		set_variable = {
			name = stabilize_british_empire_var
			value = 0
		}
	}

	option = {

		name = joi_flavor_gbr.1.a
		default_option = yes

		if = {
			limit = {
			OR = {
			NOT = { has_modifier = rule_britannia }
			NOT = { has_modifier = rise_of_british_empire }
				}
			}
			add_modifier = {
				name = rule_britannia
				months = -1				
			}
		}
		hidden_effect = {
			set_new_literacy_rate_ire = yes
			heir = {
				add_trait = persistent
				add_trait = erudite
				add_trait = aesthete
				add_trait = experienced_political_operator
			}
		}
	}
}
